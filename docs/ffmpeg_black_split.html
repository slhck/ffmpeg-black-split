165        Calculate the inverted black periods to get the content periods.

169
170        Returns:
171            list: List of content periods.
172        """
173        content_periods: list[Union[Period, OpenPeriod]] = []

177
178        for black_period in sorted_black_periods:
179            if black_period["start"] > previous_period_end:

233    def filter_black_periods(
234        black_periods: list[Period], num_cuts: int = 1
235    ) -> list[Period]:
236        """
237        Filter black periods based on desired number of cuts or custom black periods.
238
239        Args:
240            num_cuts (int): Desired number of cuts (default: 1)
241
242        Returns:
243            list[Period]: Filtered list of black periods
244        """
245        if not black_periods:
246            raise ValueError(
247                "No black periods detected. Run detect_black_periods first."
248            )
249
250        if num_cuts >= len(black_periods):
251            return black_periods
252
253        # select periods closest to evenly dividing the video
254        video_duration = max(period["end"] for period in black_periods)
255        ideal_cut_times = [
256            video_duration * (i + 1) / (num_cuts + 1) for i in range(num_cuts)
257        ]
258
259        selected_periods = []
260        for cut_time in ideal_cut_times:
261            closest_period = min(
262                black_periods,
263                key=lambda p: abs(p["start"] + p["duration"] / 2 - cut_time),
264            )
265            selected_periods.append(closest_period)
266            black_periods.remove(closest_period)
267
268        return selected_periods
